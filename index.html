HDU多校3-C

发表于 2018-07-31 | 分类于 ACM

链接

http://acm.hdu.edu.cn/showproblem.php?pid=6321

题意

给定一个 n 个点的无向图，m 次加边或者删边操作。 在每次操作后统计有多少个匹配包含 $k = 1,2,…, n/2$ 条边。 $2 ≤n≤ 10$,$1 ≤m≤ 30000$。

官方题解

  • 设 $f[i][S]$ 表示前i次操作之后，S集合的点已经匹配的方案数。
    对于加边操作，显然$f[i][S]=f[i−1][S]+f[i−1][S−u−v]$。
  • i 这一维可以省略，从大到小遍历 S，$f[S]+= f[S−u−v]$。
  • 对于删边操作，注意到加边操作的顺序不影响结果，可以假设第$i−1$ 次操作是加入要删除的边。将加边操作的更新倒过来，得到：从小到大遍历 S， $f[S]− = f[S−u−v]$。
  • 时间复杂度 O($m2^n$)。

思路

直接实现就可以了，据说爆搜会T，直接取模也会T。

牛客多校2-G

发表于 2018-07-27 | 分类于 ACM

题意

数轴上有n个集装箱，第i个集装箱位于坐标x[i]，有a[i]件货物。 现在要把集装箱进行一些移动，求在所有货物移动总距离不超过T的情况下，最多能把多少个集装箱移动到同一个位置。

官方题解

因为我们要让货物移动总距离尽可能小，所以最后所使用的集装箱的初始位置在数轴上一定是一段区间。 如果固定了这个区间，那么最优方案就是把这些集装箱移动到这些集装箱的坐标中位数的位置。
答案满足可二分性，先二分答案。然后我们按照从左至右的顺序枚举区间的左端点，那么区间的右端点和 区间的中位数都是单调递增的，一遍枚举一遍维护即可。 复杂度$O(n*log(sum(a[i])))$

题解

总体思路：

处理前缀后缀，二分答案。

前后缀：

用于处理将[l,r]区间的货品全部移动到l或者r的花费。

可行性：

二分中枚举左端点，找右端点，当左端点增加时，右端点必然右移，所以满足单调性能够在O（n）的复杂度内check一次。
左右区间确定之后，需要找出中位数，中位数也满足单调性，故不会太多的增加复杂度
通过中位数计算总的cost判断能否满足条件

注意：

由于两个端点的值不一定能够取完，一个方向的枚举只能判断左端点或者右端点没有完全使用完，因此需要两个方向的枚举，从左到右枚举和从右往左枚举能够解决。

#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;
typedef long long LL;
#define lson rt << 1, l, mid
#define rson rt << 1|1, mid + 1, r
const int MAXN = 2000 + 5;
int  Col[MAXN << 2], n, cnt, res;
double X[MAXN << 2], Sum[MAXN << 2];
struct seg {
    double l,r,h;
    int s;                                        //1或者-1 标记入边or出边
    seg() {}
    seg(double l,double r,double h,int s):l(l),r(r),h(h),s(s) {}
    bool operator < (const seg & object) const {
        return h < object.h;
    }
} S[MAXN];


void pushup(int rt,int l,int r) {
    if (Col[rt]) Sum[rt] = X[r+1] - X[l];//利用[ , ),这个区间性质，左闭右开，未完全覆盖
    else if (l == r) Sum[rt] = 0;
    else Sum[rt] = Sum[rt<<1] + Sum[rt<<1|1];  //常规操作
}

void update(int L, int R, int c,int rt,int l, int r) {
    if(L <= l && r <= R) {                     //在区间内 Cover到了这条边
        Col[rt] += c;
        pushup(rt,l,r);                        //更新sum数组，sum维护的是覆盖长度
        return ;
    }
    int mid = (l + r) >> 1;                  //递归更新子区间
    if(L <= mid) update(L, R, c, rt << 1, l, mid);
    if(R > mid) update(L, R, c, rt << 1|1, mid + 1, r);
    pushup(rt,l,r);
}

int binary_find(double x){               //之前对数据进行了离散化，现在需要找出对应的下标
    int lb = -1,ub = res - 1;
    while(ub - lb > 1){
        int mid = (lb + ub) >> 1;
        if(X[mid] >= x) ub = mid;
        else lb = mid;
    }
    return ub;
}

int main() {
    int cas = 1;
    while(~ scanf("%d", &n), n) {
        cnt = res = 0;
        for(int i = 0 ; i < n; i ++) {
            double a,b,c,d;
            scanf("%lf%lf%lf%lf",&a, &b, &c,&d);
            S[cnt] = seg(a, c, b, 1);               //水品扫描线
            X[cnt ++] = a;
            S[cnt] = seg(a, c, d, -1);
            X[cnt ++] = c;
        }
        sort(X, X + cnt);
        sort(S, S + cnt);
        res ++;
        for(int i = 1; i < cnt; i ++) {          //去重
            if(X[i] != X[i - 1]) X[res ++] = X[i];
        }

        memset(Sum, 0, sizeof(Sum));
        memset(Col, 0, sizeof(Col));
        double ans = 0;
        for(int i = 0;i < cnt - 1;i ++){
            int l = binary_find(S[i].l);
            int r = binary_find(S[i].r) - 1;//利用[ , ),这个区间性质，左闭右开
            update(l, r, S[i].s, 1, 0, res - 1);
            ans += Sum[1] * (S[i + 1].h - S[i].h);  //sum1???
        }
        printf("Test case #%d\nTotal explored area: %.2lf\n\n",cas++ , ans);
    }
    return 0;
}

#include <iostream>
#include <bits/stdc++.h>
using namespace std;

struct Line{
    double x,y1,y2;
    int val;
    Line(){}
    Line(double x,double y1,double y2,int val):x(x),y1(y1),y2(y2),val(val){};
}s[2005];
int cov[2005<<2];
double y[2005<<2],sum[2005<<2];

bool cmp(const Line &a,const Line &b)
{
    return a.x<b.x;
}
void push_up(int root,int l,int r)
{
    if(cov[root]) sum[root]=y[r+1]-y[l];
    else if(l==r) sum[root]=0;
    else sum[root]=sum[root<<1]+sum[root<<1|1];
}
void update(int L,int R,int val,int root,int l,int r)
{
    if(L<=l&&r<=R)
    {
        cov[root]+=val;
        push_up(root,l,r);
        return ;
    }
    int mid=(l+r)>>1;
    if(L<=mid) update(L,R,val,root<<1,l,mid);
    if(mid<R) update(L,R,val,root<<1|1,mid+1,r);
    push_up(root,l,r);
}
int binary_find(double yy,int tot)
{
    int ll=-1,rr=tot-1;
    while(rr-ll>1)
    {
        int mid=(ll+rr)>>1;
        if(y[mid]>=yy) rr=mid;
        else ll=mid;
    }
    return rr;
}
int main()
{
    int cas=1;
    int n;
    while(~scanf("%d",&n)&&n){
        for(int i=0;i<n;i++){
            double a,b,c,d;
            scanf("%lf%lf%lf%lf",&a,&b,&c,&d);
            s[i*2]=Line(a,b,d,1);
            s[i*2+1]=Line(c,b,d,-1);
            y[i*2]=b;
            y[i*2+1]=d;
        }
        int tot=0;
        sort(s,s+n*2,cmp);
        sort(y,y+n*2);
        tot++; //?
        for(int i=1;i<n*2;i++){
            if(y[i]!=y[i-1]) y[tot++]=y[i];
        }
        double ans=0;
        memset(sum,0,sizeof sum);
        memset(cov,0,sizeof cov);
        for(int i=0;i<n*2-1;i++)
        {
            int l=binary_find(s[i].y1,tot);
            int r=binary_find(s[i].y2,tot)-1;
            update(l,r,s[i].val,1,0,tot-1);
            ans+=sum[1]*(s[i+1].x-s[i].x);
        }
        printf("Test case #%d\nTotal explored area: %.2lf\n\n",cas++ , ans);
    }
    return 0;
}

理解：

sum[1] 维护了当前区间线段的总长度
cover用来标记矩形的覆盖情况，cover[i]标记了线段树的一个区间的线段的覆盖情况，当cover[i]>=1说明有矩形被未完全覆盖，由于线段树的特点，假如区间是[1,3][4,5],那么其中的（3,4）就未被计算在内，会出现错误，所以要利用左开右闭的特点把这个区间计算进线段树里面。
